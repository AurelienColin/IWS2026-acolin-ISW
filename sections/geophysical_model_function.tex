\begin{frame}{Geophysical Model Function}
  The normalized radar cross section $\sigma_0$ is modeled as:

  \vspace{0.5em}

  \begin{equation}
    \sigma_0(\theta, \phi, U_{10}) =
      A_0(\theta, U_{10}) \left[
        1 + A_1(\theta, U_{10}) \cos(\phi)
        + A_2(\theta, U_{10}) \cos(2\phi)
      \right]
  \end{equation}

  \vspace{0.5em}

  where:
  \begin{itemize}
    \item $\theta$ is the radar incidence angle
    \item $\phi$ is the wind direction relative to the antenna look direction
    \item $U_{10}$ is the 10-meter wind speed (m/s)
    \item $A_0, A_1, A_2$ are the harmonic coefficients
  \end{itemize}
\end{frame}
