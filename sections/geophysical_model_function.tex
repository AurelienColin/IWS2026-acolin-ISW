\begin{frame}{Equation Example}
  A general quadratic function is expressed as:

  \vspace{0.5em}

  \begin{equation}
    f(x) = ax^2 + bx + c
  \end{equation}

  \vspace{0.5em}

  where:
  \begin{itemize}
    \item $a$ is the leading coefficient
    \item $b$ is the linear coefficient
    \item $c$ is the constant term
    \item $x$ is the independent variable
  \end{itemize}
\end{frame}
